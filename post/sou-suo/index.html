#include<bits/stdc++.h>
using namespace std;

int n,m,t,sx,sy,fx,fy,cnt;
bool vis[10][10];
bool mp[10][10];
int x[]={0,0,1,-1};
int y[]={1,-1,0,0};

void dfs(int sx,int sy)
{
	if(sx==fx&&sy==fy)
	{
		cnt++;
		return;
	}
	for(int i=0;i<4;i++)
	{
		int dx = sx + x[i];
		int dy = sy + y[i];
		if(!vis[dx][dy]&&dx>=1&&dx<=n&&dy>=1&&dy<=m&&!mp[dx][dy])
		{
			vis[dx][dy]=1;
			dfs(dx,dy);
			vis[dx][dy]=0;
		}
	}
}
int main()
{
	cin>>n>>m>>t>>sx>>sy>>fx>>fy;
	while(t--)
	{
		int x,y;
		cin>>x>>y;
		mp[x][y]=1;
	}
	vis[sx][sy]=1;
	dfs(sx,sy);
	cout<<cnt;
}

int dx[4]={0,0,1,-1};
int dy[4]={1,-1,0,0};

int n,m,a[201][201],s[201][201],ans;

int dfs(int x,int y){
    if(s[x][y])return s[x][y];//è®°å¿†åŒ–æœç´¢
    s[x][y]=1;//é¢˜ç›®ä¸­ç­”æ¡ˆæ˜¯æœ‰åŒ…å«è¿™ä¸ªç‚¹çš„
